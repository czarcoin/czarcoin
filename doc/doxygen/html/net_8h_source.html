358  void AskFor(const CInv& inv)
359  {
360  // We're using mapAskFor as a priority queue,
361  // the key is the earliest time the request can be sent
362  int64 nRequestTime;
363  limitedmap<CInv, int64>::const_iterator it = mapAlreadyAskedFor.find(inv);
364  if (it != mapAlreadyAskedFor.end())
365  nRequestTime = it->second;
366  else
367  nRequestTime = 0;
368  if (fDebugNet)
369  printf("askfor %s %"PRI64d" (%s)\n", inv.ToString().c_str(), nRequestTime, DateTimeStrFormat("%H:%M:%S", nRequestTime/1000000).c_str());
370 
371  // Make sure not to reuse time indexes to keep things in the same order
372  int64 nNow = (GetTime() - 1) * 1000000;
373  static int64 nLastTime;
374  ++nLastTime;
375  nNow = std::max(nNow, nLastTime);
376  nLastTime = nNow;
377 
378  // Each retry is 2 minutes after the last
379  nRequestTime = std::max(nRequestTime + 2 * 60 * 1000000, nNow);
380  if (it != mapAlreadyAskedFor.end())
381  mapAlreadyAskedFor.update(it, nRequestTime);
382  else
383  mapAlreadyAskedFor.insert(std::make_pair(inv, nRequestTime));
384  mapAskFor.insert(std::make_pair(nRequestTime, inv));
385  }

432  memcpy((char*)&ssSend[CMessageHeader::CHECKSUM_OFFSET], &nChecksum, sizeof(nChecksum));
