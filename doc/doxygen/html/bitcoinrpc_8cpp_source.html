935 void ServiceConnection(AcceptedConnection *conn)
936 {
937  bool fRun = true;
938  while (fRun)
939  {
940  int nProto = 0;
941  map<string, string> mapHeaders;
942  string strRequest, strMethod, strURI;
943 
944  // Read HTTP request line
945  if (!ReadHTTPRequestLine(conn->stream(), nProto, strMethod, strURI))
946  break;
947 
948  // Read HTTP message headers and body
949  ReadHTTPMessage(conn->stream(), mapHeaders, strRequest, nProto);
950 
951  if (strURI != "/") {
952  conn->stream() << HTTPReply(HTTP_NOT_FOUND, "", false) << std::flush;
953  break;
954  }
955 
956  // Check authorization
957  if (mapHeaders.count("authorization") == 0)
958  {
959  conn->stream() << HTTPReply(HTTP_UNAUTHORIZED, "", false) << std::flush;
960  break;
961  }
962  if (!HTTPAuthorized(mapHeaders))
963  {
964  printf("ThreadRPCServer incorrect password attempt from %s\n", conn->peer_address_to_string().c_str());
965  /* Deter brute-forcing short passwords.
966  If this results in a DOS the user really
967  shouldn't have their RPC port exposed.*/
968  if (mapArgs["-rpcpassword"].size() < 20)
969  MilliSleep(250);
970 
971  conn->stream() << HTTPReply(HTTP_UNAUTHORIZED, "", false) << std::flush;
972  break;
973  }
974  if (mapHeaders["connection"] == "close")
975  fRun = false;
976 
977  JSONRequest jreq;
978  try
979  {
980  // Parse request
981  Value valRequest;
982  if (!read_string(strRequest, valRequest))
983  throw JSONRPCError(RPC_PARSE_ERROR, "Parse error");
984 
985  string strReply;
986 
987  // singleton request
988  if (valRequest.type() == obj_type) {
989  jreq.parse(valRequest);
990 
991  Value result = tableRPC.execute(jreq.strMethod, jreq.params);
992 
993  // Send reply
994  strReply = JSONRPCReply(result, Value::null, jreq.id);
995 
996  // array of requests
997  } else if (valRequest.type() == array_type)
998  strReply = JSONRPCExecBatch(valRequest.get_array());
999  else
1000  throw JSONRPCError(RPC_PARSE_ERROR, "Top-level object parse error");
1001 
1002  conn->stream() << HTTPReply(HTTP_OK, strReply, fRun) << std::flush;
1003  }
1004  catch (Object& objError)
1005  {
1006  ErrorReply(conn->stream(), objError, jreq.id);
1007  break;
1008  }
1009  catch (std::exception& e)
1010  {
1011  ErrorReply(conn->stream(), JSONRPCError(RPC_PARSE_ERROR, e.what()), jreq.id);
1012  break;
1013  }
1014  }
