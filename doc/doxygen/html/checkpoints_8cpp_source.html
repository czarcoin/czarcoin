81  // Guess how far we are in the verification process at the given block index
82  double GuessVerificationProgress(CBlockIndex *pindex) {
83  if (pindex==NULL)
84  return 0.0;
85 
86  int64 nNow = time(NULL);
87 
88  double fWorkBefore = 0.0; // Amount of work done before pindex
89  double fWorkAfter = 0.0; // Amount of work left after pindex (estimated)
90  // Work is defined as: 1.0 per transaction before the last checkoint, and
91  // fSigcheckVerificationFactor per transaction after.
92 
93  const CCheckpointData &data = Checkpoints();
94 
95  if (pindex->nChainTx <= data.nTransactionsLastCheckpoint) {
96  double nCheapBefore = pindex->nChainTx;
97  double nCheapAfter = data.nTransactionsLastCheckpoint - pindex->nChainTx;
98  double nExpensiveAfter = (nNow - data.nTimeLastCheckpoint)/86400.0*data.fTransactionsPerDay;
99  fWorkBefore = nCheapBefore;
100  fWorkAfter = nCheapAfter + nExpensiveAfter*fSigcheckVerificationFactor;
101  } else {
102  double nCheapBefore = data.nTransactionsLastCheckpoint;
103  double nExpensiveBefore = pindex->nChainTx - data.nTransactionsLastCheckpoint;
104  double nExpensiveAfter = (nNow - pindex->nTime)/86400.0*data.fTransactionsPerDay;
105  fWorkBefore = nCheapBefore + nExpensiveBefore*fSigcheckVerificationFactor;
106  fWorkAfter = nExpensiveAfter*fSigcheckVerificationFactor;
107  }
108 
109  return fWorkBefore / (fWorkBefore + fWorkAfter);
110  }
