114 bool CCoinsViewDB::GetStats(CCoinsStats &stats) {
115  leveldb::Iterator *pcursor = db.NewIterator();
116  pcursor->SeekToFirst();
117 
118  CHashWriter ss(SER_GETHASH, PROTOCOL_VERSION);
119  stats.hashBlock = GetBestBlock()->GetBlockHash();
120  ss << stats.hashBlock;
121  int64 nTotalAmount = 0;
122  while (pcursor->Valid()) {
123  boost::this_thread::interruption_point();
124  try {
125  leveldb::Slice slKey = pcursor->key();
126  CDataStream ssKey(slKey.data(), slKey.data()+slKey.size(), SER_DISK, CLIENT_VERSION);
127  char chType;
128  ssKey >> chType;
129  if (chType == 'c') {
130  leveldb::Slice slValue = pcursor->value();
131  CDataStream ssValue(slValue.data(), slValue.data()+slValue.size(), SER_DISK, CLIENT_VERSION);
132  CCoins coins;
133  ssValue >> coins;
134  uint256 txhash;
135  ssKey >> txhash;
136  ss << txhash;
137  ss << VARINT(coins.nVersion);
138  ss << (coins.fCoinBase ? 'c' : 'n');
139  ss << VARINT(coins.nHeight);
140  stats.nTransactions++;
141  for (unsigned int i=0; i<coins.vout.size(); i++) {
142  const CTxOut &out = coins.vout[i];
143  if (!out.IsNull()) {
144  stats.nTransactionOutputs++;
145  ss << VARINT(i+1);
146  ss << out;
147  nTotalAmount += out.nValue;
148  }
149  }
150  stats.nSerializedSize += 32 + slValue.size();
151  ss << VARINT(0);
152  }
153  pcursor->Next();
154  } catch (std::exception &e) {
155  return error("%s() : deserialize error", __PRETTY_FUNCTION__);
156  }
157  }
158  delete pcursor;
159  stats.nHeight = GetBestBlock()->nHeight;
160  stats.hashSerialized = ss.GetHash();
161  stats.nTotalAmount = nTotalAmount;
162  return true;
163 }
